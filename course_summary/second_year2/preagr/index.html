    import math

# 定义常量
PI = 3.14159265358979323846
a = 6378135.0  # 赤道半径
b = 6356750.0  # 极半径

# 度分秒转换为弧度（考虑方向）
def dms_to_radians(degrees, minutes, seconds, direction):
    decimal = degrees + minutes / 60.0 + seconds / 3600.0
    if direction in ('S', 'W'):
        decimal = -decimal
    return decimal * PI / 180.0

# 计算 Kx 系数
def calculate_Kx(lat0_rad):
    e_sq = 1 - (b * b) / (a * a)  # e²
    denom = math.sqrt(1 - e_sq * math.sin(lat0_rad) ** 2)
    return (a * math.cos(lat0_rad)) / denom

# 计算 Ky 系数
def calculate_Ky(lat0_rad):
    e_sq = 1 - (b * b) / (a * a)
    denom = (1 - e_sq * math.sin(lat0_rad) ** 2) ** 1.5
    return (a * (1 - e_sq)) / denom

# 输入第一个点（起点）的经纬度
lat_input = input().split()
lat1_degrees = float(lat_input[0])
lat1_minutes = float(lat_input[1])
lat1_seconds = float(lat_input[2])
lat1_direction = lat_input[3]

lon_input = input().split()
lon1_degrees = float(lon_input[0])
lon1_minutes = float(lon_input[1])
lon1_seconds = float(lon_input[2])
lon1_direction = lon_input[3]

# 输入第二个点的经纬度
lat_input = input().split()
lat2_degrees = float(lat_input[0])
lat2_minutes = float(lat_input[1])
lat2_seconds = float(lat_input[2])
lat2_direction = lat_input[3]

lon_input = input().split()
lon2_degrees = float(lon_input[0])
lon2_minutes = float(lon_input[1])
lon2_seconds = float(lon_input[2])
lon2_direction = lon_input[3]

# 转换为弧度
lat0 = dms_to_radians(lat1_degrees, lat1_minutes, lat1_seconds, lat1_direction)
lon0 = dms_to_radians(lon1_degrees, lon1_minutes, lon1_seconds, lon1_direction)
lat = dms_to_radians(lat2_degrees, lat2_minutes, lat2_seconds, lat2_direction)
lon = dms_to_radians(lon2_degrees, lon2_minutes, lon2_seconds, lon2_direction)

# 计算系数
Kx = calculate_Kx(lat0)
Ky = calculate_Ky(lat0)

# 计算距离
delta_lon = lon - lon0
delta_lat = lat - lat0
X = Kx * delta_lon  # 东西方向距离
Y = Ky * delta_lat  # 南北方向距离

# 输出结果
print(f" dongxi distance is {abs(X):.1f}m ")
print(f" nanbei distance is {abs(Y):.1f}m ")

    import numpy as np
from geopy.distance import geodesic

# WGS-84 椭球参数
a = 6378137  # 长半轴 (m)
b = 6356752  # 短半轴 (m)
e = np.sqrt(1 - (b / a) ** 2)  # 偏心率

def N(a, e, lat):
    """计算卯酉圈半径"""
    return a / np.sqrt(1 - e**2 * np.sin(lat)**2)

def GCS_to_ECEF(lat, lon, h=0):
    """大地坐标系 (WGS-84) 转换为 ECEF 直角坐标"""
    lat = np.radians(lat)
    lon = np.radians(lon)
    N_val = N(a, e, lat)

    X = (N_val + h) * np.cos(lat) * np.cos(lon)
    Y = (N_val + h) * np.cos(lat) * np.sin(lon)
    Z = ((1 - e**2) * N_val + h) * np.sin(lat)

    return np.array([X, Y, Z])

def ECEF_to_NED(X, Y, Z, X0, Y0, Z0, lat0, lon0):
    """ECEF 转换为 NED (局部东北坐标)"""
    dX = X - X0
    dY = Y - Y0
    dZ = Z - Z0

    lat0 = np.radians(lat0)
    lon0 = np.radians(lon0)

    # 计算旋转矩阵
    R = np.array([
        [-np.sin(lat0) * np.cos(lon0), -np.sin(lat0) * np.sin(lon0), np.cos(lat0)],
        [-np.sin(lon0), np.cos(lon0), 0],
        [-np.cos(lat0) * np.cos(lon0), -np.cos(lat0) * np.sin(lon0), -np.sin(lat0)]
    ])

    ned = R @ np.array([dX, dY, dZ])
    return ned  # 返回 (North, East, Down)

def calculate_CEP_RMS_2DRMS(errors):
    """计算 CEP, RMS, 2DRMS """
    mu = np.mean(errors)
    sigma = np.std(errors)
    CEP = mu + 0.68 * sigma
    RMS = mu + sigma
    TwoDRMS = mu + 2 * sigma
    return mu, sigma, CEP, RMS, TwoDRMS

def convert_to_decimal(value_str, direction):
    """支持两种输入格式的转换：度分秒和十进制"""
    if not value_str or not direction:
        return None

    # 检查数值合法性（不允许带符号）
    if value_str.startswith(('-', '+')):
        return None

    try:
        # 格式1：度分秒（通过空格分隔判断）
        if ' ' in value_str:
            parts = value_str.split()
            if len(parts) != 3:
                return None
            degrees = float(parts[0])
            minutes = float(parts[1])
            seconds = float(parts[2])
            decimal = degrees + minutes/60 + seconds/3600

        # 格式2：十进制（直接浮点数）
        else:
            decimal = float(value_str)

        # 处理方向
        if direction in ['S', 'W']:
            decimal *= -1

        return round(decimal, 8)

    except ValueError:
        return None

def input_coordinates(group_name):
    """输入一组坐标点，支持两种格式"""
    count = int(input(f"\n请输入 {group_name} 的坐标点数量 (输入0跳过): "))
    if count <= 0:
        return None

    coords = []
    for i in range(count):
        while True:
            data = input(
                f"请输入 {group_name} 坐标点{i+1} [格式: 纬度值,纬度方向(N/S),经度值,经度方向(E/W)]\n"
                "示例（两种格式）:\n"
                "度分秒 → 40 3 50.832,N,116 14 25.652,E\n"
                "十进制 → 40.06402,N,116.24046,E\n"
                "→ "
            )
            parts = data.split(',')
            if len(parts) == 4:
                lat_value, lat_dir, lon_value, lon_dir = parts
                # 验证方向
                if lat_dir not in ['N', 'S'] or lon_dir not in ['E', 'W']:
                    print("! 方向错误：纬度需为 N/S，经度需为 E/W")
                    continue
                # 转换坐标
                lat = convert_to_decimal(lat_value.strip(), lat_dir)
                lon = convert_to_decimal(lon_value.strip(), lon_dir)
                if lat is not None and lon is not None:
                    coords.append([lat, lon])
                    break
            print("! 格式错误，请按示例格式重新输入")

    return np.array(coords)

def input_reference_point():
    """输入基准点坐标"""
    while True:
        data = input(
            "请输入基准点坐标 [格式: 纬度值,纬度方向(N/S),经度值,经度方向(E/W)]\n"
            "示例（两种格式）:\n"
            "度分秒 → 40 3 50.832,N,116 14 25.652,E\n"
            "十进制 → 40.06402,N,116.24046,E\n"
            "→ "
        )
        parts = data.split(',')
        if len(parts) == 4:
            lat_value, lat_dir, lon_value, lon_dir = parts
            # 验证方向
            if lat_dir not in ['N', 'S'] or lon_dir not in ['E', 'W']:
                print("! 方向错误：纬度需为 N/S，经度需为 E/W")
                continue
            # 转换坐标
            lat = convert_to_decimal(lat_value.strip(), lat_dir)
            lon = convert_to_decimal(lon_value.strip(), lon_dir)
            if lat is not None and lon is not None:
                return np.array([lat, lon])
        print("! 格式错误，请按示例格式重新输入")

def main():
    print("===== GPS坐标转换与误差分析（支持度分秒/度分/十进制输入） =====")
    print("输入说明：")
    print("- 纬度方向必须为 N/S，经度方向必须为 E/W")
    print("- 数值部分禁止带正负号，正负由方向决定\n")

    # 选择模式
    while True:
        mode = input("请选择模式：\n1. 普通点坐标模式（计算均值作为基准）\n2. 基准点模式（直接输入基准点）\n→ ")
        if mode in ['1', '2']:
            break
        print("! 请输入 1 或 2")

    if mode == '1':
        # 原有的普通点坐标模式
        coords_start = input_coordinates("起点")
        coords_end = input_coordinates("终点")

        # 处理起点坐标
        if coords_start is not None:
            mean_lat_start, mean_lon_start = np.mean(coords_start, axis=0)
            X_start, Y_start, Z_start = GCS_to_ECEF(mean_lat_start, mean_lon_start)

            print("\n=== 起点坐标信息 ===")
            print(f"均值坐标 (Lat, Lon): ({mean_lat_start:.4f}, {mean_lon_start:.4f})")
            print(f"ECEF 坐标 (X, Y, Z): ({X_start:.4f}, {Y_start:.4f}, {Z_start:.4f}) m")

            # 计算起点组内误差
            if len(coords_start) > 1:
                print("\n=== 起点组内误差分析 ===")
                errors = []
                for i in range(len(coords_start)):
                    for j in range(i+1, len(coords_start)):
                        lat1, lon1 = coords_start[i]
                        lat2, lon2 = coords_start[j]
                        distance = geodesic((lat1, lon1), (lat2, lon2)).meters
                        errors.append(distance)

                mu, sigma, CEP, RMS, TwoDRMS = calculate_CEP_RMS_2DRMS(errors)
                print(f"组内平均距离: {mu:.4f} m")
                print(f"标准差: {sigma:.4f} m")
                print(f"CEP: {CEP:.4f} m")
                print(f"RMS: {RMS:.4f} m")
                print(f"2DRMS: {TwoDRMS:.4f} m")

        # 处理终点坐标
        if coords_end is not None:
            mean_lat_end, mean_lon_end = np.mean(coords_end, axis=0)
            X_end, Y_end, Z_end = GCS_to_ECEF(mean_lat_end, mean_lon_end)

            print("\n=== 终点坐标信息 ===")
            print(f"均值坐标 (Lat, Lon): ({mean_lat_end:.4f}, {mean_lon_end:.4f})")
            print(f"ECEF 坐标 (X, Y, Z): ({X_end:.4f}, {Y_end:.4f}, {Z_end:.4f}) m")

            # 计算终点组内误差
            if len(coords_end) > 1:
                print("\n=== 终点组内误差分析 ===")
                errors = []
                for i in range(len(coords_end)):
                    for j in range(i+1, len(coords_end)):
                        lat1, lon1 = coords_end[i]
                        lat2, lon2 = coords_end[j]
                        distance = geodesic((lat1, lon1), (lat2, lon2)).meters
                        errors.append(distance)

                mu, sigma, CEP, RMS, TwoDRMS = calculate_CEP_RMS_2DRMS(errors)
                print(f"组内平均距离: {mu:.4f} m")
                print(f"标准差: {sigma:.4f} m")
                print(f"CEP: {CEP:.4f} m")
                print(f"RMS: {RMS:.4f} m")
                print(f"2DRMS: {TwoDRMS:.4f} m")

        # 如果两组坐标均存在，则计算相对信息
        if coords_start is not None and coords_end is not None:
            # 计算 NED 坐标
            N, E, D = ECEF_to_NED(X_end, Y_end, Z_end, X_start, Y_start, Z_start, mean_lat_start, mean_lon_start)

            # 计算两种距离
            distance_NED = np.sqrt(N**2 + E**2)
            distance_geo = geodesic((mean_lat_start, mean_lon_start), (mean_lat_end, mean_lon_end)).meters

            # 计算误差（假设真实距离为地理测距结果）
            error = abs(distance_NED - distance_geo)

            # 计算统计指标（示例中使用单次误差）
            mu, sigma, CEP, RMS, TwoDRMS = calculate_CEP_RMS_2DRMS([error])

            print("\n=== 坐标相对关系 ===")
            print(f"NED 局部坐标 (N, E, D): ({N:.4f}, {E:.4f}, {D:.4f}) m")
            print(f"NED 平面距离: {distance_NED:.4f} m")
            print(f"地理测距结果: {distance_geo:.4f} m")
            print(f"误差值: {error:.4f} m")

            print("\n=== 误差指标 (基于单次测量) ===")
            print(f"CEP: {CEP:.4f} m")
            print(f"RMS: {RMS:.4f} m")
            print(f"2DRMS: {TwoDRMS:.4f} m")

    else:
        # 基准点模式
        print("\n=== 基准点模式 ===")
        ref_point = input_reference_point()
        ref_lat, ref_lon = ref_point
        X_ref, Y_ref, Z_ref = GCS_to_ECEF(ref_lat, ref_lon)

        print("\n=== 基准点信息 ===")
        print(f"基准坐标 (Lat, Lon): ({ref_lat:.4f}, {ref_lon:.4f})")
        print(f"ECEF 坐标 (X, Y, Z): ({X_ref:.4f}, {Y_ref:.4f}, {Z_ref:.4f}) m")

        # 输入测量点
        coords = input_coordinates("测量点")
        if coords is not None:
            print("\n=== 测量点相对误差分析 ===")
            errors = []
            for lat, lon in coords:
                # 计算相对于基准点的距离
                distance = geodesic((ref_lat, ref_lon), (lat, lon)).meters
                errors.append(distance)

            mu, sigma, CEP, RMS, TwoDRMS = calculate_CEP_RMS_2DRMS(errors)
            print(f"平均距离: {mu:.4f} m")
            print(f"标准差: {sigma:.4f} m")
            print(f"CEP: {CEP:.4f} m")
            print(f"RMS: {RMS:.4f} m")
            print(f"2DRMS: {TwoDRMS:.4f} m")

if __name__ == '__main__':
    main()
